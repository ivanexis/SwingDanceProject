CREATE TABLE CHAT_MESSAGES (
    MESSAGE_ID INT IDENTITY(3001,1) NOT NULL PRIMARY KEY,
    MATCH_ID INT NOT NULL, -- 關聯到 MATCHES 表的 MATCH_ID
    SENDER_ID INT NOT NULL, -- 發送訊息的使用者 ID
    RECEIVER_ID INT NOT NULL, -- 接收訊息的使用者 ID
    CONTENT_ID INT NOT NULL, -- 關聯到 MESSAGE_CONTENTS 表的 CONTENT_ID
    TIMESTAMP DATETIME DEFAULT GETDATE() NOT NULL, -- 訊息發送時間

CREATE TABLE MESSAGE_CONTENTS (
    CONTENT_ID INT IDENTITY(4001,1) NOT NULL PRIMARY KEY,
    MESSAGE_CONTENT NVARCHAR(MAX) NOT NULL -- 訊息內容
);
